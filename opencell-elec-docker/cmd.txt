  stage('Init Postgres keycloak DB') {
            when { expression { params.DEPLOY_MODE.startsWith('init_env') && false} }
            steps {
                script {
                    def status = sh(script: "docker inspect -f '{{.State.Status}}' postgres-wildfly || echo 'not_found'", returnStdout: true).trim()
                    
                    if (status == 'exited' || status == 'created') {
                        echo "Le conteneur postgres-wildfly est arrêté. Démarrage..."
                        sh "docker start postgres-wildfly"
                        sh "sleep 10"
                    } else if (status == 'running') {
                        echo "Le conteneur postgres-wildfly est déjà en cours d'exécution."
                    } else {
                        error("Conteneur postgres-wildfly introuvable ou dans un état inconnu: ${status}")
                    }

                    echo "Création utilisateur et base keycloak..."

                    // Création du fichier SQL temporaire
                    writeFile file: 'init_keycloak.sql', text: '''\
                        CREATE USER keycloak WITH PASSWORD 'keycloak';
                        CREATE DATABASE keycloak;
                        GRANT ALL PRIVILEGES ON DATABASE keycloak TO keycloak;
                        ALTER USER keycloak WITH SUPERUSER;
                        '''

                    // Copie dans le conteneur
                    sh 'docker cp init_keycloak.sql postgres-wildfly:/tmp/init_keycloak.sql'

                    // Exécution dans le conteneur
                    sh 'docker exec -i postgres-wildfly psql -U opencell_db_user -d opencell_db -f /tmp/init_keycloak.sql'

                    // Suppression du fichier dans le conteneur
                    sh 'docker exec -i postgres-wildfly rm /tmp/init_keycloak.sql'

                    echo "Redémarrage du conteneur keycloak..."
                    sh 'docker restart keycloak'

                    echo "Redémarrage du service opencell..."
                    sh 'docker restart opencell'
                }
            }
        }
